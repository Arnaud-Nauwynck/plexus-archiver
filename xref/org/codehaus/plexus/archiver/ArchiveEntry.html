63      /**
64       * @param name     the filename as it will appear in the archive. This is platform-specific
65       *                 normalized with File.separatorChar
66       * @param resource original filename
67       * @param type     FILE or DIRECTORY
68       * @param mode     octal unix style permissions
69       * @param collection
70       * @param defaultDirMode
71       */
72      private ArchiveEntry( String name, @Nonnull PlexusIoResource resource, int type, int mode,
73                            PlexusIoResourceCollection collection, int defaultDirMode )
74      {
75          this.name = name;
76          this.defaultDirMode = defaultDirMode;
77          try {
78              this.resource = collection != null ? collection.resolve(resource) : resource;
79          } catch (IOException e) {
80              throw new   ArchiverException("Error resolving resource " + resource.getName(), e);
81          }
82          this.attributes = ( resource instanceof ResourceAttributeSupplier)
83              ? ( (ResourceAttributeSupplier) resource ).getAttributes() : null;
84          this.type = type;
85          int permissions = mode;
86  
87          if ( mode == -1 && this.attributes == null )
88          {
89              permissions = resource.isFile() ? Archiver.DEFAULT_FILE_MODE
90                  : resource.isSymbolicLink() ? Archiver.DEFAULT_SYMLILNK_MODE : Archiver.DEFAULT_DIR_MODE;
91          }
92  
93          this.mode = permissions == -1 ? permissions : ( permissions & UnixStat.PERM_MASK ) |
94              ( type == FILE ? UnixStat.FILE_FLAG : type == SYMLINK ? UnixStat.LINK_FLAG : UnixStat.DIR_FLAG );
95      }

137     /**
138      * @return octal user/group/other unix like permissions.
139      */
140     public int getMode()
141     {
142         if ( mode != -1 )
143         {
144             return mode;
145         }
146         
147         if ( attributes != null && attributes.getOctalMode() > -1 )
148         {
149             return attributes.getOctalMode();
150         }
151         
152         return ( ( type == FILE ? Archiver.DEFAULT_FILE_MODE
153             : type == SYMLINK ? Archiver.DEFAULT_SYMLILNK_MODE : Archiver.DEFAULT_DIR_MODE ) & UnixStat.PERM_MASK ) |
154             ( type == FILE ? UnixStat.FILE_FLAG : type == SYMLINK ? UnixStat.LINK_FLAG : UnixStat.DIR_FLAG );
155     }

157     public static ArchiveEntry createFileEntry( String target, PlexusIoResource resource, int permissions,
158                                                 PlexusIoResourceCollection collection, int defaultDirectoryPermissions )
159         throws ArchiverException
160     {
161         if ( resource.isDirectory() )
162         {
163             throw new ArchiverException( "Not a file: " + resource.getName() );
164         }
165         final int type = resource.isSymbolicLink() ? SYMLINK : FILE;
166         return new ArchiveEntry( target, resource, type, permissions, collection, defaultDirectoryPermissions );
167     }

169     public static ArchiveEntry createFileEntry( String target, File file, int permissions, int defaultDirectoryPermissions )
170             throws ArchiverException, IOException {
171         if ( !file.isFile() )
172         {
173             throw new ArchiverException( "Not a file: " + file );
174         }
175         
176         final PlexusIoResource res =  ResourceFactory.createResource( file );
177 
178         final int type;
179         if (res.isSymbolicLink()){
180             type = SYMLINK;
181             permissions =  permissions & ~(UnixStat.FILE_FLAG); // remove file flag again .doh.
182         } else {
183             type = FILE; // File flag was there already. This is a bit of a mess !
184         }
185 
186         return new ArchiveEntry( target, res, type, permissions, null, defaultDirectoryPermissions );
187     }

189     public static ArchiveEntry createDirectoryEntry( String target, @Nonnull PlexusIoResource resource, int permissions,
190                                                      int defaultDirectoryPermissions )
191         throws ArchiverException
192     {
193         if ( !resource.isDirectory() )
194         {
195             throw new ArchiverException( "Not a directory: " + resource.getName() );
196         }
197         final int type;
198         if (resource.isSymbolicLink()){
199             type = SYMLINK;
200             permissions =  permissions & ~(UnixStat.DIR_FLAG); // remove dir flag again .doh.
201         } else {
202             type = DIRECTORY; // Dir flag was there already. This is a bit of a mess !
203 
204         }
205         return new ArchiveEntry( target, resource, type, permissions, null, defaultDirectoryPermissions );
206     }

208     public static ArchiveEntry createDirectoryEntry( String target, final File file, int permissions,
209                                                      int defaultDirMode1 )
210             throws ArchiverException, IOException {
211         if ( !file.isDirectory() )
212         {
213             throw new ArchiverException( "Not a directory: " + file );
214         }

220     public static ArchiveEntry createSymlinkEntry( String symlinkName, int permissions, String symlinkDestination,
221                                                    int defaultDirectoryPermissions
222     )
223     {
224 		File symlinkFile = new File(symlinkName);
225 		final ArchiveEntry archiveEntry = new ArchiveEntry(symlinkName, new PlexusIoVirtualSymlinkResource(symlinkFile, symlinkDestination), SYMLINK, permissions,
