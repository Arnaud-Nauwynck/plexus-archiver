77      private void testTarFile( Compressor compressor, String extension,
78                                TarFileCreator tarFileCreator )
79          throws Exception
80      {
81          File file = new File( "target/output/TarFileTest.tar" );
82          final TarArchiver archiver = (TarArchiver) lookup( Archiver.ROLE, "tar" );
83  		archiver.setLongfile(TarLongFileMode.posix );
84          archiver.setDestFile( file );
85          archiver.addDirectory( new File( "src" ) );
86          FileUtils.removePath( file.getPath() );
87          archiver.createArchive();
88          if ( compressor != null )
89          {
90              final File compressedFile = new File( file.getPath() + extension );
91  			compressor.setSource( createResource( file, file.getName() ) );
92              compressor.setDestFile( compressedFile );
93              compressor.compress();
94              compressor.close();
95              file = compressedFile;
96          }
97          final TarFile tarFile = tarFileCreator.newTarFile( file );
98  
99          for ( Enumeration en = tarFile.getEntries();  en.hasMoreElements();  )
100         {
101             final TarArchiveEntry te = (TarArchiveEntry) en.nextElement();
102             if ( te.isDirectory() || te.isSymbolicLink())
103             {
104                 continue;
105             }
106             final File teFile = new File( "src", te.getName() );
107             final InputStream teStream = tarFile.getInputStream( te );
108             final InputStream fileStream = new FileInputStream( teFile );
109             assertTrue( Arrays.equals( IOUtil.toByteArray( teStream ), IOUtil.toByteArray( fileStream ) ) );
110             teStream.close();
111             fileStream.close();
112         }
113         
114         tarFile.close();
115     }
