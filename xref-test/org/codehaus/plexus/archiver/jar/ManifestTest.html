237     public void checkMultiLineAttribute( String in, String expected )
238         throws Exception
239     {
240         StringWriter writer = new StringWriter();
241         Manifest.Attribute attr = new Manifest.Attribute();
242         attr.setName( "test" );
243         attr.setValue( in );
244         attr.write( writer );
245         writer.flush();
246 
247         // Print the string with whitespace replaced with special codes
248         // so in case of failure you can see what went wrong.
249         System.err.println( "String: " + dumpString( writer.toString() ) );
250 
251         assertEquals( "should be indented multiline", "test: " + expected, writer.toString() );
252     }

254     private static String dumpString( String in )
255     {
256         String out = "";
257 
258         char[] chars = in.toCharArray();
259 
260         for ( char aChar : chars )
261         {
262             switch ( aChar )
263             {
264                 case '\t':
265                     out += "\\t";
266                     break;
267                 case '\r':
268                     out += "\\r";
269                     break;
270                 case '\n':
271                     out += "\\n";
272                     break;
273                 case ' ':
274                     out += "\\s";
275                     break;
276                 default:
277                     out += aChar;
278                     break;
279             }
280         }
281 
282         return out;
283     }
