54          Random rand = new Random();
55          for ( int i = 0; i < 45000; i++ )
56          {
57             File f = new File( tmpDir, "file" + i );
58             f.deleteOnExit();
59             FileOutputStream out = new FileOutputStream(f);
60              byte[] data = new byte[512]; // 512bytes per file
61             rand.nextBytes( data );
62             out.write( data );
63             out.flush();
64             out.close();
65          }
