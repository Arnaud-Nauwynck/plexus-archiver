26      {

38       * Creates a map with the archive files contents. The map keys
39       * are the names of file entries in the archive file. The map
40       * values are the respective archive entries.
41       */
42      private static Map<String, TarArchiveEntry> getFileEntries( ArchiveFile file )
43          throws IOException
44      {
45          final Map map = new HashMap();
46          for ( java.util.Enumeration en = file.getEntries();  en.hasMoreElements();  )
47          {
48              TarArchiveEntry ze = (TarArchiveEntry) en.nextElement();
49              if ( ze.isDirectory() )
50              {
51                  continue;
52              }
53              if ( map.put( ze.getName(), ze ) != null )
54              {
55                  Assert.fail( "Multiple archive file entries named " + ze.getName() + " found." );
56              }
57          }
58          return map;
59      }
